[hear] (define test-graph / make-graph (vector 1 2 3 4) (vector (vector 1 2) (vector 2 3) (vector 1 4)));

[hear] (define graph-linked / lambda (g n1 n2) (exists / ? idx / if (and (>= (idx) 0) (< (idx) (list-length / list-ref (g) 1))) (list= (list-ref (list-ref (g) 1) (idx)) (vector (n1) (n2))) (false)));

[hear] (= (graph-linked (test-graph) 1 2) (true));

[hear] (= (graph-linked (test-graph) 1 3) (false));

[hear] (= (graph-linked (test-graph) 2 4) (false));
